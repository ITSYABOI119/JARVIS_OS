# JARVIS Phase 1 - seL4 Tutorial Integration
cmake_minimum_required(VERSION 3.7.2)
project(hello-world C ASM)

# Include tutorial helpers (provided by init script)
include(${SEL4_TUTORIALS_DIR}/settings.cmake)

# Let tutorial framework regenerate to set up kernel targets
sel4_tutorials_regenerate_tutorial(${CMAKE_CURRENT_SOURCE_DIR})

# Use tutorial framework's app declaration macro
DeclareTutorialApp(
    hello-world
    SOURCES
        src/main.c
        src/cache/decision_cache.c
        src/cache/cache_patterns.c
        src/ipc/ring_buffer.c
        src/ipc/ipc_sel4.c
    INCLUDES
        src
        src/cache
        src/ipc
    LIBS
        sel4
        muslc
        utils
        sel4muslcsys
        sel4platsupport
        sel4utils
        sel4debug
)
